const chaser = document.getElementById("chaser");

const isMobile = window.matchMedia("(pointer: coarse)").matches;

let mouseX = window.innerWidth / 2;
let mouseY = window.innerHeight / 2;

let posX = mouseX;
let posY = mouseY;

let targetX = mouseX;
let targetY = mouseY;

let frame = 0;
let lastTime = 0;
const frameRate = 150;

let state = "wander";
let stateTimer = Date.now();

const STATES = {
    WANDER: "wander",
    CHASE: "chase",
    ATTACK: "attack",
    IDLE: "idle"
};

function randomTarget() {
    targetX = Math.random() * (window.innerWidth - 150);
    targetY = Math.random() * (window.innerHeight - 150);
}

randomTarget();

window.addEventListener("mousemove", (e) => {

    mouseX = e.clientX;
    mouseY = e.clientY;

    if (!isMobile) {

        const dist = Math.hypot(
            mouseX - posX,
            mouseY - posY
        );

        if (dist < 400 && Math.random() < 0.01) {
            state = STATES.CHASE;
            stateTimer = Date.now();
        }
    }
});

window.addEventListener("scroll", () => {

    if (isMobile) {

        state = STATES.ATTACK;

        clearTimeout(window.attackTimeout);

        window.attackTimeout = setTimeout(() => {

            state = STATES.WANDER;
            randomTarget();

        }, 1500);
    }
});

function update(time) {

    const now = Date.now();

    if (!isMobile) {

        if (
            state === STATES.CHASE &&
            now - stateTimer > 5000
        ) {
            state = STATES.WANDER;
            randomTarget();
        }

        if (
            state === STATES.WANDER &&
            now - stateTimer > 4000
        ) {

            const roll = Math.random();

            if (roll < 0.2) {

                state = STATES.CHASE;

            } else if (roll < 0.4) {

                state = STATES.IDLE;

                setTimeout(() => {
                    state = STATES.WANDER;
                    randomTarget();
                    stateTimer = Date.now();
                }, 1500);

            } else {

                randomTarget();
            }

            stateTimer = now;
        }
    }

    switch (state) {

        case STATES.WANDER:

            posX += (targetX - posX) * 0.015;
            posY += (targetY - posY) * 0.015;

            if (
                Math.hypot(
                    targetX - posX,
                    targetY - posY
                ) < 25
            ) {
                randomTarget();
            }

            break;

        case STATES.CHASE:

            posX += (mouseX - posX) * 0.025;
            posY += (mouseY - posY) * 0.025;

            const chaseDistance =
                Math.hypot(
                    mouseX - posX,
                    mouseY - posY
                );

            if (chaseDistance < 100) {

                state = STATES.ATTACK;

                setTimeout(() => {

                    state = STATES.WANDER;
                    randomTarget();

                }, 1000);
            }

            break;

        case STATES.ATTACK:

            posX += (mouseX - posX) * 0.08;
            posY += (mouseY - posY) * 0.08;

            break;

        case STATES.IDLE:
            break;
    }

    let lookX = targetX;
    let lookY = targetY;

    if (
        state === STATES.CHASE ||
        state === STATES.ATTACK
    ) {
        lookX = mouseX;
        lookY = mouseY;
    }

    const angle =
        Math.atan2(
            lookY - posY,
            lookX - posX
        ) * 180 / Math.PI + 70;

    chaser.style.transform =
        `translate(${posX}px, ${posY}px) rotate(${angle}deg)`;

    if (time - lastTime > frameRate) {

        frame = (frame + 1) % 2;

        chaser.style.backgroundPosition =
            `-${frame * 150}px 0`;

        lastTime = time;
    }

    requestAnimationFrame(update);
}

requestAnimationFrame(update);
